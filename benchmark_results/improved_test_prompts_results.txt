```python
    """Check if a number is prime."""

    # Edge cases
    if n <= 1:
        return False
    elif n == 2:
```

```python
from typing import List

def second_largest(numbers: List[int]) -> int:
    """Find the second largest number in a list of numbers"""

    if len(numbers) < 2:
        return None  # Return None for edge case where there are less than two elements

    first_max = second_max = float('-inf')

    for num in numbers:
        if num > first_max:
            second_max = first_max
            first_max = num
        elif num > second_max and num != first_max:  # Avoid duplicate of the max value
            second_max = num

    return second_max

if __name__ == "__main__":
    print(second_largest([1, 2, 3]))  # Expected output: 2
    print(second_largest([5, 4, 1, 2, 3]))  # Expected output: 4
    print(second_largest([9]))  # Expected output: None (as there is only one element)
```

```python
def bubble_sort(numbers: list) -> list:
    """Bubble sort implementation"""
    # Get the length of the numbers list
    n = len(numbers)

    # Iterate over each number in the list
    for i in range(n):
        # Last i elements are already sorted, so we iterate from 0 to n-i-1
        for j in range(0, n - i - 1):
            # If current element is greater than next one, swap them
            if numbers[j] > numbers[j + 1]:
                numbers[j], numbers[j + 1] = numbers[j + 1], numbers[j]

    return numbers

if __name__ == "__main__":
    print(bubble_sort([64, 34, 25, 12, 22, 11, 90]))
```

```python
def least_common_multiple(a: int, b: int) -> int:
    """

    Parameters:
        a (int): First number.
        b (int): Second number.

    Returns:
        lcm (int): Least Common Multiple of a and b.
    """
    # Handle edge cases where either or both inputs are 0
    if a == 0 and b == 0:
        return 0

    def gcd(x: int, y: int) -> int:
        while y:
            x, y = y, x % y
        return x

    lcm = abs(a*b) // gcd(a, b) if a and b else 0

    return lcm
```

```python
from typing import List

def quicksort(arr: List[int]) -> List[int]:
    """Implements the quicksort algorithm."""

    if len(arr) <= 1:
        return arr
    else:
        pivot = arr.pop()

    items_greater = []
    items_lower = []

    for item in arr:
        if item > pivot:
            items_greater.append(item)
        else:
            items_lower.append(item)

    return quicksort(items_lower) + [pivot] + quicksort(items_greater)

if __name__ == "__main__":
```

```python
def merge_sort(arr):
    """

    :param arr: List of unsorted elements
    :return: Sorted list
    """
    if len(arr) <= 1:  # base case
        return arr

    mid = len(arr) // 2
    left_half = merge_sort(arr[:mid])
    right_half = merge_sort(arr[mid:])

    return merge(left_half, right_half)

def merge(left, right):
    """

    :param left: Left half of the array
    :param right: Right half of the array
    :return: Merged and sorted list
    """
    merged = []
    while left and right:  # merge smaller elements first
        if left[0] <= right[0]:  
            merged.append(left[0])
            left = left[1:]
        else:
            merged.append(right[0])
            right = right[1:]
    while left:  # add remaining elements from the left half
        merged.append(left[0])
        left = left[1:]
    while right:  # add remaining elements from the right half
        merged.append(right[0])
        right = right[1:]

    return merged

if __name__ == "__main__":

    test_arr = [38, 27, 43, 3, 9, 82, 10]

    print("Sorted array:", sorted_arr)
```

```python
    """Check if parentheses in a string are balanced."""
    stack = []
    for char in s:
        if char == '(':

            if not stack or stack[-1] != '(':
```

```python
def fibonacci_sequence(n: int) -> int:
    """
    >>> fibonacci_sequence(10)
    55
    """
    if n < 0:
        raise ValueError("Input 0 or negative is not valid.")
    cache = [None] * (n + 1)
    return helper(n, cache)

def helper(n: int, cache: list) -> int:
    if n <= 1:
        return n
    elif cache[n] != None:
        return cache[n]
    else:
        result = helper(n-1, cache) + helper(n-2, cache)
        cache[n] = result
        return result
```
